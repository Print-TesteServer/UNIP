Exibir o primeiro nome do funcionário, exibir o nome do departamento de todos os funcionários.

SELECT E.FIRST_NAME, D.DEPARTMENT_NAME 
FROM EMPLOYEES E, DEPARTMENTS D
WHERE E.DEPARTMENT_ID = D.DEPARTMENT_ID

#Exibir o ultimo nome dos funcionários, o nome do cargo, o nome do departamento, nome da rua, o nome do país, o nome da região para todos os funcionários.

SELECT E.LAST_NAME AS Sobrenome , J.JOB_TITLE AS Cargo, D.DEPARTMENT_NAME AS Departamento, L.STREET_ADDRESS AS Endereco, C.COUNTRY_NAME AS Pais , R.REGION_NAME AS Regiao
FROM EMPLOYEES E, JOBS J, DEPARTMENTS D, LOCATIONS L, COUNTRIES C, REGIONS R
WHERE E.JOB_ID = J.JOB_ID
AND   E.DEPARTMENT_ID = D.DEPARTMENT_ID
AND   D.LOCATION_ID = L.LOCATION_ID
AND   L.COUNTRY_ID = C.COUNTRY_ID
AND   C.REGION_ID = R.REGION_ID


#Exibir o primeiro nome dos funcionários, o código do departamento, o código do gerente, o salário e o cargo para os funcionários com cargos de “Programmer” ou “Sales Manager”.

SELECT E.FIRST_NAME, E.DEPARTMENT_ID, (SELECT E1.FIRST_NAME FROM EMPLOYEES E1 WHERE E.MANAGER_ID = E1.EMPLOYEE_ID) AS Gerente , E.SALARY, J.JOB_TITLE
FROM EMPLOYEES E, JOBS J
WHERE E.JOB_ID = J.JOB_ID
AND J.JOB_TITLE = 'Programmer'
OR  J.JOB_TITLE = 'Sales Manager'

#Exibir o código do funcionário, o primeiro nome, o código do departamento, o código do gerente e o salário para funcionários com o primeiro nome de “John” ou “Peter” ou “Alana”.

SELECT E.EMPLOYEE_ID, E.FIRST_NAME, D.DEPARTMENT_NAME, E.MANAGER_ID, E.SALARY
FROM EMPLOYEES E, DEPARTMENTS D
WHERE E.FIRST_NAME = 'John'
OR E.FIRST_NAME = 'Peter'
OR E.FIRST_NAME = 'Alana'
AND E.DEPARTMENT_ID = D.DEPARTMENT_ID


#Exibir todos os funcionários que tenham o primeiro nome iniciado com a letra “J”.
#Exibir todos os funcionários que tenham o primeiro nome terminado com a letra “s”.

SELECT *
FROM EMPLOYEES
WHERE FIRST_NAME LIKE 'J%'

SELECT *
FROM EMPLOYEES
WHERE FIRST_NAME LIKE '%s'

#Exibir todas as regiões de todos os departamentos.

SELECT D.DEPARTMENT_NAME AS Departamento, R.REGION_NAME AS Pais
FROM DEPARTMENTS D, COUNTRIES C, LOCATIONS L, REGIONS R
WHERE D.LOCATION_ID = L.LOCATION_ID
AND   L.COUNTRY_ID = C.COUNTRY_ID
AND   C.REGION_ID = R.REGION_ID

#Exibir onde fica a cidade do departamento de administração.

SELECT D.DEPARTMENT_NAME, L.CITY
FROM DEPARTMENTS D, LOCATIONS L
WHERE D.LOCATION_ID = L.LOCATION_ID
AND   D.DEPARTMENT_NAME = 'Administration'

#mostrar salarios maximos e minimos

SELECT JOB_TITLE as Cargo, MIN_SALARY as Salario_Minimo, MAX_SALARY AS Salario_Maximo
FROM JOBS

#média de todos os cargos

SELECT AVG(MAX_SALARY)
FROM JOBS
